## SAWTOOTH COMMANDS ##
#######################

## Get your public key for adding to GitHub
cat ~/.ssh/id_rsa.pub

## Copy/paste the output from the command above into a new deploy key for your transaction processor's GitHub repo

## Clone your transaction processor repo
cd ~ && git clone git@github.com:__YOUR_TRANSACTION_PROCESSOR_GITHUB_REPO__.git

## Put sawtooth-core in your transaction processor
ln -s ~/sawtooth-core ~/__YOUR_TRANSACTION_PROCESSOR_REPO_NAME__/sawtooth-core

## Build your transaction processor
cd ~/__YOUR_TRANSACTION_PROCESSOR_REPO_NAME__/ && npm install && cd ~

## Start Sawtooth services
sudo systemctl start sawtooth-validator && sudo systemctl start sawtooth-rest-api && sudo systemctl start sawtooth-intkey-tp-python && sudo systemctl start sawtooth-settings-tp && sudo systemctl start sawtooth-identity-tp

## Confirm Sawtooth services started successfully
sudo journalctl -f -u sawtooth-validator -u sawtooth-rest-api -u sawtooth-intkey-tp-python -u sawtooth-settings-tp -u sawtooth-identity-tp

## Allow yo'self to be yo'self
sawset proposal create --key ~/.sawtooth/keys/sawtooth.priv --url http://0.0.0.0:8008 sawtooth.identity.allowed_keys=`cat ~/.sawtooth/keys/sawtooth.pub`

## Add permitted devices by public key
## THIS DOESN'T WORK because of whatever
# sawtooth identity policy create --key ~/.sawtooth/keys/sawtooth.priv --url http://0.0.0.0:8008 permitted_devices "PERMIT_KEY `cat ~/.sawtooth/keys/sawtooth.pub`"

## Add all permitted devices
## THIS WORKS because it just allows everyone
sawtooth identity policy create --key ~/.sawtooth/keys/sawtooth.priv --url http://0.0.0.0:8008 permitted_devices "PERMIT_KEY *"

## Add permitted devices by public key
## THIS NEEDS TO BE DONE IN A DIFFERENT WAY
## Because otherwise it overwrites your previous policy rather than appending to it
# sawtooth identity policy create --key ~/.sawtooth/keys/sawtooth.priv --url http://0.0.0.0:8008 permitted_devices "PERMIT_KEY __PUBLIC_KEY_OF_PERMITTED_DEVICE__"

## create a transactor role for permitted devices
sawtooth identity role create --key ~/.sawtooth/keys/sawtooth.priv --url http://0.0.0.0:8008 transactor permitted_devices

## Test some transactions
## Create a batch of intkey transactions
intkey create_batch

## Load that intkey batch onto the blockchain
intkey load

## See those hot, hot blocks
sawtooth block list
