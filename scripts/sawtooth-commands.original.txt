## Get your public key for adding to GitHub
cat /home/sawtooth/.ssh/id_rsa.pub

## Copy/paste the output from the command above into your transaction processor's GitHub repo

## Clone your transaction processor repo
cd /home/sawtooth
git clone __YOUR_TRANSACTION_PROCESSOR_GITHUB_REPO__

## Build sawtooth-core
cd /home/sawtooth/sawtooth-core && git checkout tags/v1.0.0rc6 -b tags/v1.0.0rc6 && cd sdk/javascript && npm install && cd /home/sawtooth

## Put sawtooth-core's Javascript SDK in your transaction processor
cp -R /home/sawtooth/sawtooth-core/sdk/javascript /home/sawtooth/__YOUR_TRANSACTION_PROCESSOR_REPO_NAME__/

## Build your transaction processor
cd /home/sawtooth/__YOUR_TRANSACTION_PROCESSOR_REPO_NAME__/ && yarn

## Start Sawtooth services
sudo systemctl start sawtooth-validator && sudo systemctl start sawtooth-rest-api && sudo systemctl start sawtooth-intkey-tp-python && sudo systemctl start sawtooth-settings-tp && sudo systemctl start sawtooth-identity-tp

## Confirm Sawtooth services started successfully
sudo journalctl -f -u sawtooth-validator -u sawtooth-rest-api -u sawtooth-intkey-tp-python -u sawtooth-settings-tp -u sawtooth-identity-tp

## Allow yo'self to be yo'self
sawset proposal create --key /home/sawtooth/.sawtooth/keys/sawtooth.priv --url http://0.0.0.0:8008 sawtooth.identity.allowed_keys=`cat /home/sawtooth/.sawtooth/keys/sawtooth.pub`

## Add permitted devices by public key
sawtooth identity policy create --key /home/sawtooth/.sawtooth/keys/sawtooth.priv --url http://0.0.0.0:8008 permitted_devices "PERMIT_KEY 
__PUBLIC_KEY_OF_PERMITTED_DEVICE__"

## create a transactor role for permitted devices
sawtooth identity role create --key /home/sawtooth/.sawtooth/keys/sawtooth.priv --url http://0.0.0.0:8008 transactor permitted_devices

## Test some transactions
## Create a batch of intkey transactions
intkey create_batch

## Load that intkey batch onto the blockchain
intkey load

## See those hot, hot blocks
sawtooth block list
